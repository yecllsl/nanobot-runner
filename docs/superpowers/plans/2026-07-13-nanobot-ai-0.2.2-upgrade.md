**Tech Stack:** Python 3.11+、nanobot-ai 0.2.2、FastAPI、Polars、Typer、pytest、Playwright

```python
# src/core/config_injector.py
"""配置注入器 - 通过 SDK 配置注入替代 monkey-patch

替代 _patch_websocket_settings_api() 机制，将 RunFlowAgent 自有配置
（~/.nanobot-runner/config.json）注入 nanobot 运行时。
"""
import json
import logging
from pathlib import Path

from typing import Any

logger = logging.getLogger(__name__)


class ConfigInjectionError(Exception):
    """配置注入失败异常"""


class ConfigInjector:
    """通过 SDK 配置注入替代 monkey-patch

    将 RunFlowAgent 配置（~/.nanobot-runner/config.json）转换为
    nanobot Config 对象，替代运行时 patch load_config/save_config。
    """

    def __init__(self, config_path: Path):
        """初始化配置注入器

        Args:
            config_path: RunFlowAgent 配置文件路径
        """
        self.config_path = config_path

    def build_nanobot_config(self, runner_config: dict[str, Any]) -> Any:
        """构建 nanobot Config 对象

        Args:
            runner_config: RunFlowAgent 配置字典

        Returns:
            Config: nanobot 配置对象

        Raises:
            ConfigInjectionError: 配置构建失败
        """
        try:
            from nanobot.config.schema import (
                AgentsConfig,
                AgentDefaults,
                Config,
                ProvidersConfig,
                WebSocketConfig,
                WebUIConfig,
                ToolsConfig,
            )

            agents_defaults = runner_config.get("agents", {}).get("defaults", {})
            providers_raw = runner_config.get("providers", {})
            websocket_raw = runner_config.get("websocket", {})
            webui_raw = runner_config.get("webui", {})
            tools_raw = runner_config.get("tools", {})

            # 构建 Provider 配置
            providers = {}
            for name, prov_cfg in providers_raw.items():
                providers[name] = {
                    "api_key": prov_cfg.get("api_key", ""),
                    "api_base": prov_cfg.get("api_base", ""),
                }

            config = Config(
                agents=AgentsConfig(
                    defaults=AgentDefaults(
                        model=agents_defaults.get("model", "gpt-4"),
                        provider=agents_defaults.get("provider", "openai"),
                        timezone=agents_defaults.get("timezone", "UTC"),
                        workspace=agents_defaults.get("workspace", "~/.nanobot-runner/"),
                        max_tool_iterations=agents_defaults.get("max_tool_iterations", 10),
                        context_window_tokens=agents_defaults.get("context_window_tokens", 128000),
                        context_block_limit=agents_defaults.get("context_block_limit", 50000),
                        max_tool_result_chars=agents_defaults.get("max_tool_result_chars", 30000),
                    )
                ),
                providers=ProvidersConfig(**providers) if providers else ProvidersConfig(),
                websocket=WebSocketConfig(**websocket_raw) if websocket_raw else WebSocketConfig(),
                webui=WebUIConfig(**webui_raw) if webui_raw else WebUIConfig(),
                tools=ToolsConfig(**tools_raw) if tools_raw else ToolsConfig(),
            )

            logger.debug("成功构建 nanobot Config")
            return config

        except Exception as e:
            logger.error(f"配置注入失败: {e}")
            raise ConfigInjectionError(
                f"无法构建 nanobot Config: {e}。请检查 ~/.nanobot-runner/config.json"
            ) from e

    def save_runner_config(self, config: dict[str, Any]) -> None:
        """保存 RunFlowAgent 配置

        Args:
            config: 配置字典
        """
        self.config_path.parent.mkdir(parents=True, exist_ok=True)
        with open(self.config_path, "w", encoding="utf-8") as f:
            json.dump(config, f, indent=2, ensure_ascii=False)
        logger.debug(f"配置已保存到 {self.config_path}")

    def resolve_webui_dist(self) -> Path | None:
        """解析 WebUI dist 目录

        优先 RunFlowAgent 自有 dist，回退 nanobot dist。

        Returns:
            Path | None: dist 目录路径，不存在返回 None
        """
        # 优先 RunFlowAgent 自有 dist
        custom_dist = Path(__file__).parent.parent.parent / "webui" / "dist"
        if custom_dist.exists():
            return custom_dist

        # 回退 nanobot dist
        try:
            import nanobot.web as web_pkg
            nanobot_dist = Path(web_pkg.__file__).parent / "dist"
            if nanobot_dist.exists():
                return nanobot_dist
        except (ImportError, AttributeError):
            pass

        return None
```

```python
# src/core/agent_loop_adapter.py
"""AgentLoop 适配器 - 封装私有 API 调用

隔离 nanobot 版本变更风险，为业务模块提供稳定接口。
所有 _ 前缀的私有 API 调用必须通过本 Adapter。
"""
import logging
from typing import Any

logger = logging.getLogger(__name__)


class AdapterError(Exception):
    """Adapter 操作失败异常"""


class AgentLoopAdapter:
    """封装 AgentLoop 私有 API，提供稳定接口

    优先使用 0.2.2 新增的公开 API（如 submit_cron_turn），
    无法替代的私有 API 保留调用但集中在本类内。
    """

    def __init__(self, agent_loop: Any):
        """初始化 AgentLoopAdapter

        Args:
            agent_loop: nanobot AgentLoop 实例
        """
        self._loop = agent_loop

    async def connect_mcp(self) -> None:
        """连接 MCP 服务器（封装 _connect_mcp）"""
        try:
            await self._loop._connect_mcp()
        except AttributeError as e:
            raise AdapterError(
                f"AgentLoop._connect_mcp 不可用: {e}"
            ) from e

    @property
    def mcp_stacks(self) -> list[Any]:
        """MCP 栈（封装 _mcp_stacks）"""
        try:
            return self._loop._mcp_stacks
        except AttributeError as e:
            raise AdapterError(
                f"AgentLoop._mcp_stacks 不可用: {e}"
            ) from e

    @property
    def background_tasks(self) -> set[Any]:
        """后台任务集合（封装 _background_tasks）"""
        try:
            return self._loop._background_tasks
        except AttributeError as e:
            raise AdapterError(
                f"AgentLoop._background_tasks 不可用: {e}"
            ) from e

    def add_hook(self, hook: Any) -> None:
        """添加 Hook（封装 _extra_hooks 追加）"""
        try:
            self._loop._extra_hooks.append(hook)
        except AttributeError as e:
            raise AdapterError(
                f"AgentLoop._extra_hooks 不可用: {e}"
            ) from e

    async def close_mcp(self) -> None:
        """关闭 MCP 连接"""
        await self._loop.close_mcp()

    async def process_direct(self, message: str, **kwargs: Any) -> str:
        """直接处理消息（封装 process_direct）"""
        return await self._loop.process_direct(message, **kwargs)

    async def submit_cron_turn(self, session_key: str, message: str) -> None:
        """提交 Cron turn（0.2.2 新增公开 API）"""
        await self._loop.submit_cron_turn(session_key, message)

    async def stop(self) -> None:
        """停止 Agent Loop"""
        await self._loop.stop()
```

```python
# src/core/sdk_adapter.py
"""SDK 适配器 - 编程式 Agent 调用入口

封装 nanobot 0.2.2 Python SDK，为业务模块提供编程式 Agent 调用能力，
绕过 Gateway 消息总线。
"""
import logging
from collections.abc import AsyncIterator
from typing import Any

logger = logging.getLogger(__name__)


class SDKUnavailableError(Exception):
    """SDK 不可用异常"""


class SDKAdapter:
    """封装 nanobot 0.2.2 SDK，提供编程式 Agent 调用

    使用场景：
    - 进化引擎触发 Agent 决策（无需 Gateway）
    - 数字孪生 What-If 推演（编程式调用）
    - 训练计划自动生成（嵌入业务流程）
    """

    def __init__(self, config: Any, config_injector: Any):
        """初始化 SDKAdapter

        Args:
            config: nanobot Config 对象
            config_injector: ConfigInjector 实例
        """
        self._config = config
        self._injector = config_injector

    async def create_session(self, session_key: str) -> Any:
        """创建 SDK 会话

        Args:
            session_key: 会话键

        Returns:
            NanobotSDK: 支持 async with 的 SDK 实例

        Raises:
            SDKUnavailableError: SDK 创建失败
        """
        try:
            from nanobot.nanobot import NanobotSDK

            # 创建 AgentLoop 并包装为 SDK
            # 实际实现需根据 nanobot 0.2.2 SDK API 调整
            logger.debug(f"创建 SDK 会话: {session_key}")
            return NanobotSDK(config=self._config)
        except Exception as e:
            logger.error(f"SDK 会话创建失败: {e}")
            raise SDKUnavailableError(f"SDK 模式不可用: {e}") from e

    async def stream_query(
        self, message: str, session_key: str | None = None
    ) -> AsyncIterator[str]:
        """流式查询 Agent

        Args:
            message: 用户消息
            session_key: 会话键（None 表示新会话）

        Yields:
            str: Agent 响应片段
        """
        try:
            async with await self.create_session(session_key or "default") as sdk:
                async for chunk in sdk.stream(message):
                    yield chunk
        except Exception as e:
            logger.warning(f"SDK 流式查询失败，降级到 Gateway 模式: {e}")
            raise SDKUnavailableError(
                "SDK 模式不可用，请使用 Gateway 模式"
            ) from e

    async def query(self, message: str, session_key: str | None = None) -> str:
        """同步查询 Agent（非流式）

        Args:
            message: 用户消息
            session_key: 会话键

        Returns:
            str: Agent 完整响应
        """
        result: list[str] = []
        async for chunk in self.stream_query(message, session_key):
            result.append(chunk)
        return "".join(result)
```

```python
# src/core/transparency/runtime_event_hook.py
"""运行时事件订阅 Hook - 订阅 RuntimeEventPublisher

订阅 nanobot 0.2.2 的 RuntimeEventPublisher，将运行时事件
（工具调用进度、文件编辑等）转发到 WebUI 和业务模块。
"""
import logging
from collections.abc import Callable
from dataclasses import dataclass, field
from typing import Any

from nanobot.agent.hook import AgentHook, AgentHookContext

logger = logging.getLogger(__name__)


@dataclass
class RuntimeEvent:
    """运行时事件数据结构"""
    type: str
    trace_id: str = ""
    data: Any = field(default=None)


class RuntimeEventHook(AgentHook):
    """订阅运行时事件总线，转发到 WebUI

    作为 AgentHook 子类挂在 AgentLoop 上，在 Hook 生命周期中
    发布运行时事件，支持多订阅者。
    """

    def __init__(self, event_publisher: Any):
        """初始化 RuntimeEventHook

        Args:
            event_publisher: RuntimeEventPublisher 实例
        """
        self._publisher = event_publisher
        self._subscribers: list[Callable[[RuntimeEvent], None]] = []

    def subscribe(self, callback: Callable[[RuntimeEvent], None]) -> None:
        """订阅运行时事件

        Args:
            callback: 事件回调函数
        """
        self._subscribers.append(callback)

    async def before_iteration(self, context: AgentHookContext) -> None:
        """迭代前发布事件"""
        event = RuntimeEvent(
            type="iteration_start",
            trace_id=getattr(context, "trace_id", ""),
        )
        self._publish(event)

    async def before_execute_tools(self, context: AgentHookContext) -> None:
        """工具执行前发布事件"""
        for tool_call in getattr(context, "tool_calls", []):
            event = RuntimeEvent(
                type="tool_start",
                trace_id=getattr(context, "trace_id", ""),
                data=tool_call,
            )
            self._publish(event)

    async def after_iteration(self, context: AgentHookContext) -> None:
        """迭代后发布事件"""
        event = RuntimeEvent(
            type="iteration_end",
            trace_id=getattr(context, "trace_id", ""),
        )
        self._publish(event)

    def _publish(self, event: RuntimeEvent) -> None:
        """发布事件到所有订阅者

        异常静默降级，不影响主流程。
        """
        for callback in self._subscribers:
            try:
                callback(event)
            except Exception:
                logger.exception("运行时事件订阅回调异常，已静默降级")
```

```python
class DynamicProviderRegistry:
    """动态 Provider 注册器

    利用 0.2.2 的 create_dynamic_spec() 和 ProvidersConfig.extra="allow" 能力，
    支持运行时注册自定义 OpenAI 兼容 Provider。
    """

    _custom_providers: dict[str, Any] = {}

    @classmethod
    def register_custom_provider(
        cls,
        name: str,
        api_base: str,
        api_key: str,
        default_model: str,
    ) -> None:
        """注册自定义 OpenAI 兼容 Provider

        Args:
            name: Provider 名称（不能与内置冲突）
            api_base: API 端点
            api_key: API 密钥
            default_model: 默认模型
        """
        try:
            from nanobot.providers.registry import create_dynamic_spec

            # 检查名称冲突
            builtin_providers = {"openai", "anthropic", "azure", "mistral", "bedrock"}
            if name.lower() in builtin_providers:
                logger.warning(f"Provider 名称 '{name}' 与内置冲突，拒绝注册")
                return

            spec = create_dynamic_spec(
                name=name,
                api_base=api_base,
                api_key=api_key,
                default_model=default_model,
            )
            cls._custom_providers[name] = spec
            logger.info(f"自定义 Provider '{name}' 注册成功")
        except Exception as e:
            logger.warning(f"自定义 Provider '{name}' 注册失败: {e}")

    @classmethod
    def list_custom_providers(cls) -> list[str]:
        """列出已注册的自定义 Provider"""
        return list(cls._custom_providers.keys())
```

```python
"""运行时事件 SSE 端点"""
import asyncio
import json

from fastapi import APIRouter, Request
from fastapi.responses import StreamingResponse

from src.core.transparency.runtime_event_hook import RuntimeEvent

router = APIRouter(prefix="/api/runtime-events", tags=["runtime-events"])


@router.get("/stream")
async def stream_runtime_events(request: Request):
    """SSE 流式推送运行时事件"""

    async def event_generator():
        queue: asyncio.Queue = asyncio.Queue()
        callback = lambda event: queue.put_nowait(event)

        # 订阅 RuntimeEventHook
        runtime_hook = request.app.state.runtime_event_hook
        runtime_hook.subscribe(callback)

        try:
            while True:
                if await request.is_disconnected():
                    break
                event = await queue.get()
                yield f"data: {json.dumps(event.__dict__)}\n\n"
        finally:
            runtime_hook._subscribers.remove(callback)

    return StreamingResponse(event_generator(), media_type="text/event-stream")
```
