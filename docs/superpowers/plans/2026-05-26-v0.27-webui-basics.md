```python
def get_websocket_config(self) -> dict[str, Any]:
    """读取 WebSocket 通道配置

    从 config.json 的 websocket 配置节读取，
    支持环境变量覆盖（NANOBOT_WS_*）。

    Returns:
        dict[str, Any]: WebSocket 配置字典，配置节不存在时返回空 dict
    """
    config = self.load_config()
    ws_config: dict[str, Any] = config.get("websocket", {})

    if isinstance(ws_config, dict):
        ws_config = ws_config.copy()
    else:
        ws_config = {}

    if (v := os.getenv("NANOBOT_WS_ENABLED")) is not None:
        ws_config["enabled"] = v.lower() in ("true", "1", "yes")
    if (v := os.getenv("NANOBOT_WS_HOST")) is not None:
        ws_config["host"] = v
    if (v := os.getenv("NANOBOT_WS_PORT")) is not None:
        try:
            ws_config["port"] = int(v)
        except ValueError:
            pass
    if (v := os.getenv("NANOBOT_WS_TOKEN")) is not None:
        ws_config["token"] = v
    if (v := os.getenv("NANOBOT_WS_TOKEN_SECRET")) is not None:
        ws_config["token_issue_secret"] = v

    return ws_config
```
