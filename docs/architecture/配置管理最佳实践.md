# 配置管理最佳实践指南

## 1. 概述

本文档定义了 Nanobot Runner 项目的配置管理最佳实践，旨在解决 v0.4.1 版本暴露的配置管理架构缺陷，提供统一的配置管理规范和实施指南。

## 2. 配置分层架构

### 2.1 三级配置体系

```
┌─────────────────────────────────────────────────────────────┐
│                    运行时配置 (环境变量)                      │
│  - 敏感信息 (API密钥、密码)                                  │
│  - 临时配置覆盖                                             │
│  - 最高优先级                                               │
└─────────────────────────────────────────────────────────────┘
                              ↓
┌─────────────────────────────────────────────────────────────┐
│                    应用配置 (业务配置)                        │
│  ~/.nanobot-runner/config.json                              │
│  - 业务参数 (飞书配置、数据路径)                             │
│  - 用户偏好设置                                             │
│  - 中等优先级                                               │
└─────────────────────────────────────────────────────────────┘
                              ↓
┌─────────────────────────────────────────────────────────────┐
│                    框架配置 (底座配置)                        │
│  ~/.nanobot/config.json                                     │
│  - LLM Provider 配置                                        │
│  - 全局工具链配置                                           │
│  - 最低优先级                                               │
└─────────────────────────────────────────────────────────────┘
```

### 2.2 配置优先级规则

```python
# 配置加载顺序（优先级从高到低）
1. 环境变量 (NANOBOT_RUNNER_*)
2. 应用配置 (~/.nanobot-runner/config.json)
3. 框架配置 (~/.nanobot/config.json)
4. 代码默认值
```

## 3. 配置Schema设计

### 3.1 Schema定义原则

1. **显式优于隐式**: 所有配置项必须有明确的Schema定义
2. **类型安全**: 配置项必须有明确的类型约束
3. **必填项标识**: 区分必填项和可选项
4. **默认值策略**: 合理的默认值，避免空值错误
5. **版本兼容**: 支持配置版本迁移

### 3.2 推荐Schema实现

```python
# src/core/config_schema.py
from dataclasses import dataclass, field
from typing import Optional, Any
from enum import Enum
import json


class ConfigVersion(Enum):
    """配置版本"""
    V0_4_1 = "0.4.1"
    V0_4_2 = "0.4.2"
    V0_5_0 = "0.5.0"


@dataclass
class FeishuConfig:
    """飞书配置Schema"""
    app_id: Optional[str] = None
    app_secret: Optional[str] = None
    receive_id: Optional[str] = None
    receive_id_type: str = "user_id"  # user_id, open_id, union_id
    webhook_url: Optional[str] = None  # 兼容旧版Webhook
    
    # 验证规则
    def validate(self) -> tuple[bool, list[str]]:
        """验证飞书配置"""
        errors = []
        
        # 如果使用App方式，必须配置app_id和app_secret
        if self.app_id and not self.app_secret:
            errors.append("配置feishu_app_id时，必须同时配置feishu_app_secret")
        
        # 验证receive_id_type
        valid_types = ["user_id", "open_id", "union_id", "email", "chat_id"]
        if self.receive_id_type not in valid_types:
            errors.append(f"feishu_receive_id_type必须是以下之一: {valid_types}")
        
        return len(errors) == 0, errors


@dataclass
class DataConfig:
    """数据配置Schema"""
    data_dir: str = "~/.nanobot-runner/data"
    index_file: str = "index.json"
    partition_strategy: str = "year"  # year, quarter, month
    compression: str = "snappy"  # snappy, zstd, gzip
    
    def validate(self) -> tuple[bool, list[str]]:
        """验证数据配置"""
        errors = []
        
        valid_strategies = ["year", "quarter", "month"]
        if self.partition_strategy not in valid_strategies:
            errors.append(f"partition_strategy必须是以下之一: {valid_strategies}")
        
        valid_compressions = ["snappy", "zstd", "gzip", "none"]
        if self.compression not in valid_compressions:
            errors.append(f"compression必须是以下之一: {valid_compressions}")
        
        return len(errors) == 0, errors


@dataclass
class AppConfig:
    """应用配置Schema"""
    version: str = "0.4.1"
    feishu: FeishuConfig = field(default_factory=FeishuConfig)
    data: DataConfig = field(default_factory=DataConfig)
    auto_push_feishu: bool = False
    debug_mode: bool = False
    log_level: str = "INFO"  # DEBUG, INFO, WARNING, ERROR
    
    # 必填字段
    REQUIRED_FIELDS = ["version"]
    
    @classmethod
    def from_dict(cls, config: dict) -> "AppConfig":
        """从字典创建配置对象"""
        return cls(
            version=config.get("version", "0.4.1"),
            feishu=FeishuConfig(
                app_id=config.get("feishu_app_id"),
                app_secret=config.get("feishu_app_secret"),
                receive_id=config.get("feishu_receive_id"),
                receive_id_type=config.get("feishu_receive_id_type", "user_id"),
                webhook_url=config.get("feishu_webhook"),
            ),
            data=DataConfig(
                data_dir=config.get("data_dir", "~/.nanobot-runner/data"),
            ),
            auto_push_feishu=config.get("auto_push_feishu", False),
            debug_mode=config.get("debug_mode", False),
            log_level=config.get("log_level", "INFO"),
        )
    
    def to_dict(self) -> dict:
        """转换为字典"""
        return {
            "version": self.version,
            "data_dir": self.data.data_dir,
            "auto_push_feishu": self.auto_push_feishu,
            "feishu_app_id": self.feishu.app_id,
            "feishu_app_secret": self.feishu.app_secret,
            "feishu_receive_id": self.feishu.receive_id,
            "feishu_receive_id_type": self.feishu.receive_id_type,
            "feishu_webhook": self.feishu.webhook_url,
            "debug_mode": self.debug_mode,
            "log_level": self.log_level,
        }
    
    def validate(self) -> tuple[bool, list[str]]:
        """验证完整配置"""
        errors = []
        
        # 验证必填字段
        for field in self.REQUIRED_FIELDS:
            if not getattr(self, field):
                errors.append(f"缺少必填字段: {field}")
        
        # 验证版本格式
        if self.version:
            try:
                parts = self.version.split(".")
                if len(parts) != 3:
                    errors.append("version必须是语义化版本格式 (如: 0.4.1)")
            except Exception:
                errors.append("version格式无效")
        
        # 验证子配置
        feishu_valid, feishu_errors = self.feishu.validate()
        if not feishu_valid:
            errors.extend([f"飞书配置: {e}" for e in feishu_errors])
        
        data_valid, data_errors = self.data.validate()
        if not data_valid:
            errors.extend([f"数据配置: {e}" for e in data_errors])
        
        # 验证log_level
        valid_levels = ["DEBUG", "INFO", "WARNING", "ERROR"]
        if self.log_level not in valid_levels:
            errors.append(f"log_level必须是以下之一: {valid_levels}")
        
        return len(errors) == 0, errors
```

## 4. 配置管理实现

### 4.1 配置管理器设计

```python
# src/core/config_manager.py
import os
import json
from pathlib import Path
from typing import Optional, Any
import logging

from .config_schema import AppConfig

logger = logging.getLogger(__name__)


class ConfigManager:
    """配置管理器
    
    职责：
    1. 管理配置文件的读写
    2. 提供配置验证功能
    3. 支持环境变量覆盖
    4. 支持配置迁移
    """
    
    _instance: Optional["ConfigManager"] = None
    
    def __new__(cls, *args, **kwargs):
        """单例模式"""
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(
        self,
        config_path: Optional[Path] = None,
        auto_create: bool = True,
        validate: bool = True
    ):
        """初始化配置管理器
        
        Args:
            config_path: 配置文件路径，默认使用 ~/.nanobot-runner/config.json
            auto_create: 配置文件不存在时是否自动创建
            validate: 加载时是否验证配置
        """
        # 避免重复初始化
        if hasattr(self, '_initialized'):
            return
        self._initialized = True
        
        self.base_dir = Path.home() / ".nanobot-runner"
        self.config_path = config_path or self.base_dir / "config.json"
        self._config: Optional[AppConfig] = None
        self._validate = validate
        
        # 确保目录存在
        self._ensure_dirs()
        
        # 加载配置
        if self.config_path.exists():
            self._load_config()
        elif auto_create:
            self._create_default_config()
        
        # 应用环境变量覆盖
        self._apply_env_overrides()
    
    def _ensure_dirs(self):
        """确保配置目录存在"""
        self.base_dir.mkdir(parents=True, exist_ok=True)
        (self.base_dir / "data").mkdir(exist_ok=True)
        (self.base_dir / "cron").mkdir(exist_ok=True)
    
    def _load_config(self):
        """加载配置文件"""
        try:
            with open(self.config_path, 'r', encoding='utf-8') as f:
                config_dict = json.load(f)
            
            self._config = AppConfig.from_dict(config_dict)
            
            # 验证配置
            if self._validate:
                valid, errors = self._config.validate()
                if not valid:
                    logger.warning(f"配置验证失败: {errors}")
                    # 不抛出异常，允许使用默认配置继续
            
            logger.info(f"配置加载成功: {self.config_path}")
            
        except json.JSONDecodeError as e:
            logger.error(f"配置文件格式错误: {e}")
            self._create_default_config()
        except Exception as e:
            logger.error(f"加载配置失败: {e}")
            self._create_default_config()
    
    def _create_default_config(self):
        """创建默认配置"""
        self._config = AppConfig()
        self.save_config()
        logger.info(f"已创建默认配置: {self.config_path}")
    
    def _apply_env_overrides(self):
        """应用环境变量覆盖"""
        if self._config is None:
            return
        
        # 环境变量映射
        env_mappings = {
            "NANOBOT_RUNNER_FEISHU_APP_ID": ("feishu", "app_id"),
            "NANOBOT_RUNNER_FEISHU_APP_SECRET": ("feishu", "app_secret"),
            "NANOBOT_RUNNER_FEISHU_RECEIVE_ID": ("feishu", "receive_id"),
            "NANOBOT_RUNNER_DATA_DIR": ("data", "data_dir"),
            "NANOBOT_RUNNER_LOG_LEVEL": (None, "log_level"),
            "NANOBOT_RUNNER_DEBUG_MODE": (None, "debug_mode"),
        }
        
        for env_var, (section, key) in env_mappings.items():
            value = os.getenv(env_var)
            if value is not None:
                # 布尔值转换
                if key in ["debug_mode", "auto_push_feishu"]:
                    value = value.lower() in ("true", "1", "yes", "on")
                
                if section:
                    config_section = getattr(self._config, section)
                    setattr(config_section, key, value)
                else:
                    setattr(self._config, key, value)
                
                logger.debug(f"环境变量覆盖: {env_var}")
    
    def get(self, key: str, default: Any = None) -> Any:
        """获取配置项
        
        Args:
            key: 配置项名称，支持点号分隔 (如: "feishu.app_id")
            default: 默认值
        
        Returns:
            配置值或默认值
        """
        if self._config is None:
            return default
        
        try:
            if "." in key:
                parts = key.split(".")
                value = self._config
                for part in parts:
                    value = getattr(value, part)
                return value
            else:
                return getattr(self._config, key, default)
        except AttributeError:
            return default
    
    def set(self, key: str, value: Any, persist: bool = True):
        """设置配置项
        
        Args:
            key: 配置项名称
            value: 配置值
            persist: 是否立即持久化到文件
        """
        if self._config is None:
            self._config = AppConfig()
        
        if "." in key:
            parts = key.split(".")
            target = self._config
            for part in parts[:-1]:
                target = getattr(target, part)
            setattr(target, parts[-1], value)
        else:
            setattr(self._config, key, value)
        
        if persist:
            self.save_config()
    
    def save_config(self):
        """保存配置到文件"""
        if self._config is None:
            return
        
        try:
            config_dict = self._config.to_dict()
            with open(self.config_path, 'w', encoding='utf-8') as f:
                json.dump(config_dict, f, indent=2, ensure_ascii=False)
            logger.debug(f"配置已保存: {self.config_path}")
        except Exception as e:
            logger.error(f"保存配置失败: {e}")
    
    def validate(self) -> tuple[bool, list[str]]:
        """验证当前配置"""
        if self._config is None:
            return False, ["配置未加载"]
        return self._config.validate()
    
    def reload(self):
        """重新加载配置"""
        self._load_config()
        self._apply_env_overrides()
    
    @property
    def config(self) -> Optional[AppConfig]:
        """获取配置对象"""
        return self._config
    
    @property
    def data_dir(self) -> Path:
        """获取数据目录"""
        if self._config:
            return Path(self._config.data.data_dir).expanduser()
        return self.base_dir / "data"
    
    @property
    def cron_store(self) -> Path:
        """获取定时任务存储路径"""
        return self.base_dir / "cron" / "jobs.json"
```

## 5. 配置迁移策略

### 5.1 迁移管理器设计

```python
# src/core/config_migration.py
import json
import shutil
from dataclasses import dataclass, asdict
from datetime import datetime
from pathlib import Path
from typing import Optional, Callable
import logging

logger = logging.getLogger(__name__)


@dataclass
class MigrationRecord:
    """迁移记录"""
    migration_id: str
    source: str
    target: str
    status: str  # pending, in_progress, completed, failed
    timestamp: str
    error: Optional[str] = None


class ConfigMigrationManager:
    """配置迁移管理器
    
    职责：
    1. 管理配置迁移过程
    2. 记录迁移状态
    3. 支持迁移回滚
    4. 确保迁移幂等性
    """
    
    def __init__(self, status_file: Optional[Path] = None):
        self.status_file = status_file or (
            Path.home() / ".nanobot-runner" / "migration_status.json"
        )
        self._records: dict[str, MigrationRecord] = {}
        self._load_records()
    
    def _load_records(self):
        """加载迁移记录"""
        if self.status_file.exists():
            try:
                with open(self.status_file, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    self._records = {
                        k: MigrationRecord(**v) for k, v in data.items()
                    }
            except Exception as e:
                logger.warning(f"加载迁移记录失败: {e}")
                self._records = {}
    
    def _save_records(self):
        """保存迁移记录"""
        try:
            self.status_file.parent.mkdir(parents=True, exist_ok=True)
            with open(self.status_file, 'w', encoding='utf-8') as f:
                json.dump(
                    {k: asdict(v) for k, v in self._records.items()},
                    f,
                    indent=2,
                    ensure_ascii=False
                )
        except Exception as e:
            logger.error(f"保存迁移记录失败: {e}")
    
    def migrate(
        self,
        source: Path,
        target: Path,
        transform: Optional[Callable[[dict], dict]] = None,
        verify: bool = True
    ) -> bool:
        """执行配置迁移
        
        Args:
            source: 源配置文件路径
            target: 目标配置文件路径
            transform: 配置转换函数
            verify: 是否验证迁移结果
        
        Returns:
            迁移是否成功
        """
        migration_id = f"{source.name}_{target.name}"
        
        # 检查是否已迁移
        if self.is_migrated(migration_id):
            logger.info(f"迁移已完成，跳过: {migration_id}")
            return True
        
        # 检查源文件是否存在
        if not source.exists():
            logger.info(f"源文件不存在，无需迁移: {source}")
            return True
        
        # 检查目标文件是否已存在
        if target.exists():
            logger.warning(f"目标文件已存在，跳过迁移: {target}")
            self._record_migration(migration_id, source, target, "completed")
            return True
        
        # 记录迁移开始
        self._record_migration(migration_id, source, target, "in_progress")
        
        try:
            # 确保目标目录存在
            target.parent.mkdir(parents=True, exist_ok=True)
            
            # 如果需要转换
            if transform:
                with open(source, 'r', encoding='utf-8') as f:
                    config = json.load(f)
                config = transform(config)
                with open(target, 'w', encoding='utf-8') as f:
                    json.dump(config, f, indent=2, ensure_ascii=False)
            else:
                # 直接复制
                shutil.copy2(source, target)
            
            # 验证迁移结果
            if verify and not self._verify_migration(source, target):
                raise ValueError("迁移结果验证失败")
            
            # 记录迁移完成
            self._record_migration(migration_id, source, target, "completed")
            logger.info(f"迁移成功: {source} -> {target}")
            return True
            
        except Exception as e:
            # 记录迁移失败
            self._record_migration(migration_id, source, target, "failed", str(e))
            logger.error(f"迁移失败: {e}")
            
            # 清理失败的目标文件
            if target.exists():
                try:
                    target.unlink()
                except Exception:
                    pass
            
            return False
    
    def _record_migration(
        self,
        migration_id: str,
        source: Path,
        target: Path,
        status: str,
        error: Optional[str] = None
    ):
        """记录迁移状态"""
        self._records[migration_id] = MigrationRecord(
            migration_id=migration_id,
            source=str(source),
            target=str(target),
            status=status,
            timestamp=datetime.now().isoformat(),
            error=error
        )
        self._save_records()
    
    def _verify_migration(self, source: Path, target: Path) -> bool:
        """验证迁移结果"""
        try:
            # 验证目标文件存在
            if not target.exists():
                return False
            
            # 验证文件大小（允许一定差异）
            source_size = source.stat().st_size
            target_size = target.stat().st_size
            
            # 如果进行了转换，大小可能不同，但至少应该有内容
            if target_size == 0:
                return False
            
            # 验证JSON格式
            with open(target, 'r', encoding='utf-8') as f:
                json.load(f)
            
            return True
        except Exception as e:
            logger.warning(f"验证迁移结果失败: {e}")
            return False
    
    def is_migrated(self, migration_id: str) -> bool:
        """检查是否已完成迁移"""
        record = self._records.get(migration_id)
        return record is not None and record.status == "completed"
    
    def rollback(self, migration_id: str) -> bool:
        """回滚迁移
        
        Args:
            migration_id: 迁移ID
        
        Returns:
            回滚是否成功
        """
        record = self._records.get(migration_id)
        if not record:
            logger.warning(f"未找到迁移记录: {migration_id}")
            return False
        
        target = Path(record.target)
        
        try:
            if target.exists():
                target.unlink()
                logger.info(f"已删除迁移文件: {target}")
            
            # 更新记录状态
            record.status = "rolled_back"
            record.timestamp = datetime.now().isoformat()
            self._save_records()
            
            return True
        except Exception as e:
            logger.error(f"回滚失败: {e}")
            return False
    
    def get_migration_history(self) -> list[MigrationRecord]:
        """获取迁移历史"""
        return sorted(
            self._records.values(),
            key=lambda x: x.timestamp,
            reverse=True
        )
```

### 5.2 配置版本转换

```python
# src/core/config_transforms.py
"""配置版本转换函数"""

from typing import Callable


def v040_to_v041(config: dict) -> dict:
    """v0.4.0 到 v0.4.1 的配置转换"""
    # v0.4.1 主要是添加了飞书配置字段
    if "feishu_receive_id_type" not in config:
        config["feishu_receive_id_type"] = "user_id"
    
    # 更新版本号
    config["version"] = "0.4.1"
    
    return config


def v041_to_v042(config: dict) -> dict:
    """v0.4.1 到 v0.4.2 的配置转换"""
    # v0.4.2 添加了调试配置
    if "debug_mode" not in config:
        config["debug_mode"] = False
    
    if "log_level" not in config:
        config["log_level"] = "INFO"
    
    # 更新版本号
    config["version"] = "0.4.2"
    
    return config


# 版本转换映射
VERSION_TRANSFORMS: dict[str, Callable[[dict], dict]] = {
    "0.4.0": v040_to_v041,
    "0.4.1": v041_to_v042,
}


def transform_config(config: dict, target_version: str) -> dict:
    """转换配置到目标版本
    
    Args:
        config: 当前配置
        target_version: 目标版本
    
    Returns:
        转换后的配置
    """
    current_version = config.get("version", "0.4.0")
    
    # 依次应用转换
    for version, transform in sorted(VERSION_TRANSFORMS.items()):
        if current_version < version <= target_version:
            config = transform(config)
    
    return config
```

```python
# 推荐方式：使用单例ConfigManager
from src.core.config_manager import ConfigManager

config = ConfigManager()

# 获取单个配置项
app_id = config.get("feishu.app_id")
data_dir = config.get("data.data_dir", "~/.nanobot-runner/data")

# 获取配置对象
app_config = config.config
if app_config:
    print(app_config.feishu.app_id)
```

### 6.2 更新配置

```python
from src.core.config_manager import ConfigManager

config = ConfigManager()

# 更新单个配置项（自动持久化）
```

## 8. 常见问题

### Q1: 配置文件损坏怎么办？

ConfigManager会自动创建默认配置，损坏的配置文件会被备份：

```python
# 备份损坏的配置
if config_path.exists():
    backup_path = config_path.with_suffix('.json.bak')
    shutil.move(config_path, backup_path)
    logger.warning(f"配置已备份: {backup_path}")
```

### Q2: 如何支持多环境配置？

```python
# 根据环境变量加载不同配置
env = os.getenv("NANOBOT_RUNNER_ENV", "production")
config_path = base_dir / f"config.{env}.json"
config = ConfigManager(config_path=config_path)
```

### Q3: 配置热更新如何实现？

```python
# 重新加载配置
config.reload()

# 或者在获取配置时检查文件修改时间
```
