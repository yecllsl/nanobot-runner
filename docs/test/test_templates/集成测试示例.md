```python
import pytest
from pathlib import Path
from src.core.context import get_context

class TestDataImportFlow:
    """数据导入流程集成测试"""

    @pytest.fixture
    def temp_workspace(self, tmp_path):
        """创建临时工作区"""
        return tmp_path

    def test_import_to_query_flow(self, temp_workspace):
        """测试导入到查询完整链路"""
        # Arrange
        context = get_context(workspace_dir=temp_workspace)

        # Act - 导入
        import_result = context.importer.import_file(
            Path("tests/data/fixtures/easy_run_20240101.fit")
        )

        # Act - 查询
        stats = context.analytics.get_stats()

        # Assert
        assert import_result.success
        assert stats["total_runs"] >= 1

class TestPlanningFlow:
    """智能跑步计划流程集成测试"""

    @pytest.fixture
    def temp_plan_workspace(self, tmp_path):
        """创建临时计划工作区"""
        return tmp_path

    def test_plan_generation_to_feedback_flow(self, temp_plan_workspace):
        """测试计划生成到反馈收集完整链路"""
        # Arrange
        context = get_context(workspace_dir=temp_plan_workspace)
        plan_generator = context.plan_generator
        feedback_collector = context.feedback_collector

        # Act - 生成计划
        plan = plan_generator.generate(
            target="marathon",
            target_time="3:30:00",
            weeks=12
        )

        # Act - 记录执行反馈
        feedback = feedback_collector.collect(
            plan_id=plan.id,
            session_id="session_001",
            completion_rate=0.95,
            perceived_effort=7
        )

        # Assert
        assert plan is not None
        assert len(plan.sessions) > 0
        assert feedback.completion_rate == 0.95
```
