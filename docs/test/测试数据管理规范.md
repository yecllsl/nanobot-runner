## 一、规范概述

### 1.1 目的

本规范旨在统一测试数据的管理方式，确保测试数据的质量、安全性和可维护性，为自动化测试提供可靠的数据支撑。

### 1.2 适用范围

- 单元测试数据
- 集成测试数据
- E2E测试数据
- 性能测试数据

### 1.3 基本原则

1. **数据隔离**: 测试数据与生产数据完全隔离
2. **版本控制**: 测试数据纳入版本控制
3. **数据脱敏**: 禁止使用真实用户数据
4. **可追溯性**: 测试数据来源清晰可追溯

## 四、测试数据生成

### 4.1 生成脚本规范

```python
# tests/scripts/generate_test_data.py
"""
测试数据生成脚本

使用方式:
    uv run python tests/scripts/generate_test_data.py
"""

import json
import hashlib
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any


class TestDataGenerator:
    """测试数据生成器"""
    
    def __init__(self, output_dir: Path):
        self.output_dir = output_dir
        self.output_dir.mkdir(parents=True, exist_ok=True)
        self.metadata = {
            "version": "1.0",
            "generated_at": datetime.now().isoformat(),
            "generator": __file__,
            "files": []
        }
    
    def generate_activity_metadata(
        self,
        count: int = 100,
        start_date: datetime = None
    ) -> list[dict[str, Any]]:
        """生成活动元数据"""
        import random
        
        if start_date is None:
            start_date = datetime(2024, 1, 1)
        
        activities = []
        for i in range(count):
            date = start_date + timedelta(days=i)
            
            activity = {
                "activity_id": f"test_act_{i:04d}",
                "timestamp": date.isoformat(),
                "filename": f"test_run_{date.strftime('%Y%m%d')}.fit",
                "total_distance": round(random.uniform(3000, 21000), 2),
                "total_timer_time": round(random.uniform(1200, 7200), 0),
                "avg_heart_rate": random.randint(120, 170),
                "max_heart_rate": random.randint(160, 190),
                "avg_cadence": random.randint(160, 180),
                "tags": self._generate_tags()
            }
            activities.append(activity)
        
        return activities
    
    def _generate_tags(self) -> list[str]:
        """生成随机标签"""
        import random
        all_tags = ["easy", "tempo", "interval", "long", "recovery", "race"]
        return random.sample(all_tags, k=random.randint(1, 2))
    
    def calculate_checksum(self, filepath: Path) -> str:
        """计算文件校验和"""
        sha256_hash = hashlib.sha256()
        with open(filepath, "rb") as f:
            for byte_block in iter(lambda: f.read(4096), b""):
                sha256_hash.update(byte_block)
        return f"sha256:{sha256_hash.hexdigest()}"
    
    def save_metadata(self):
        """保存元数据文件"""
        metadata_file = self.output_dir / "test_files_metadata.json"
        with open(metadata_file, "w", encoding="utf-8") as f:
            json.dump(self.metadata, f, indent=2, ensure_ascii=False)
        print(f"元数据已保存: {metadata_file}")


def main():
    """主函数"""
    # 设置输出目录
    output_dir = Path(__file__).parent.parent / "data" / "fixtures"
    
    # 创建生成器
    generator = TestDataGenerator(output_dir)
    
    # 生成测试数据
    print("开始生成测试数据...")
    activities = generator.generate_activity_metadata(count=50)
    
    # 保存活动数据
    activities_file = output_dir / "generated_activities.json"
    with open(activities_file, "w", encoding="utf-8") as f:
        json.dump(activities, f, indent=2, ensure_ascii=False)
    
    # 更新元数据
    generator.metadata["files"].append({
        "filename": "generated_activities.json",
        "type": "json",
        "size_bytes": activities_file.stat().st_size,
        "checksum": generator.calculate_checksum(activities_file),
        "description": "生成的测试活动数据",
        "record_count": len(activities),
        "tags": ["generated", "activities"]
    })
    
    # 保存元数据
    generator.save_metadata()
    
    print(f"测试数据生成完成，共 {len(activities)} 条记录")


if __name__ == "__main__":
    main()
```

```python
# 测试用例中使用测试数据示例
import json
from pathlib import Path
import pytest


class TestDataLoader:
    """测试数据加载器"""
    
    BASE_DIR = Path(__file__).parent.parent / "data"
    
    @classmethod
    def load_fixture(cls, filename: str) -> dict:
        """加载测试固件"""
        filepath = cls.BASE_DIR / "fixtures" / filename
        with open(filepath, "r", encoding="utf-8") as f:
            return json.load(f)
    
    @classmethod
    def load_sample(cls, filename: str) -> dict:
        """加载样本数据"""
        filepath = cls.BASE_DIR / "samples" / filename
        with open(filepath, "r", encoding="utf-8") as f:
            return json.load(f)
    
    @classmethod
    def get_expected_result(cls, test_name: str) -> dict:
        """获取预期结果"""
        filepath = cls.BASE_DIR / "validation" / "expected_results.json"
        with open(filepath, "r", encoding="utf-8") as f:
            data = json.load(f)
            return data.get(test_name, {})


# 使用示例
@pytest.fixture
def sample_activity():
    """样本活动数据fixture"""
    return TestDataLoader.load_fixture("generated_activities.json")[0]
```

### 5.2 数据隔离原则

```python
# 确保测试数据隔离的fixture
@pytest.fixture(scope="function")
def isolated_data_dir(tmp_path):
    """提供隔离的数据目录"""
    # 创建临时数据目录
    data_dir = tmp_path / "data"
    data_dir.mkdir()
    
    # 复制必要的测试数据
    import shutil
    fixtures_dir = Path(__file__).parent.parent / "data" / "fixtures"
    for file in fixtures_dir.glob("*.json"):
        shutil.copy(file, data_dir)
    
    # 设置环境变量
    import os
    old_data_dir = os.environ.get("NANOBOT_RUNNER_DATA_DIR")
    os.environ["NANOBOT_RUNNER_DATA_DIR"] = str(data_dir)
    
    yield data_dir
    
    # 恢复环境变量
    if old_data_dir:
        os.environ["NANOBOT_RUNNER_DATA_DIR"] = old_data_dir
    else:
        del os.environ["NANOBOT_RUNNER_DATA_DIR"]
```

## 六、数据安全规范

### 6.1 禁止事项

- ❌ 禁止使用真实用户数据
- ❌ 禁止提交包含敏感信息的测试数据
- ❌ 禁止在测试数据中硬编码密码或密钥
- ❌ 禁止共享个人设备上的测试数据

### 6.2 脱敏要求

```python
def anonymize_data(data: dict) -> dict:
    """数据脱敏函数"""
    sensitive_fields = ["name", "email", "phone", "address", "user_id"]
    
    anonymized = data.copy()
    for field in sensitive_fields:
        if field in anonymized:
            anonymized[field] = f"[REDACTED_{field}]"
    
    return anonymized
```

### 6.3 数据验证

```python
def validate_test_data(filepath: Path) -> bool:
    """验证测试数据"""
    # 检查文件大小
    if filepath.stat().st_size > 100 * 1024 * 1024:  # 100MB
        raise ValueError(f"文件过大: {filepath}")
    
    # 检查文件类型
    allowed_extensions = {".json", ".parquet", ".fit", ".csv"}
    if filepath.suffix not in allowed_extensions:
        raise ValueError(f"不支持的文件类型: {filepath.suffix}")
    
    # 检查敏感信息
    content = filepath.read_text()
    sensitive_patterns = ["password", "secret", "key", "token"]
    for pattern in sensitive_patterns:
        if pattern in content.lower():
            raise ValueError(f"发现敏感信息: {pattern}")
    
    return True
```
