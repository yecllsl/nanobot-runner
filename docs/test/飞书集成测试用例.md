```python
def test_cron_config_migration():
    """测试定时任务配置迁移"""
    from pathlib import Path
    import json
    
    old_path = Path.home() / ".nanobot" / "cron" / "jobs.json"
    new_path = Path.home() / ".nanobot-runner" / "cron" / "jobs.json"
    
    # 创建旧配置
    old_path.parent.mkdir(parents=True, exist_ok=True)
    old_config = {"jobs": [{"name": "morning_report", "schedule": "0 7 * * *"}]}
    old_path.write_text(json.dumps(old_config))
    
    # 触发迁移
    from src.core.config import ConfigManager
    config = ConfigManager()
    config._migrate_cron_config()
    
    # 验证新配置存在
    assert new_path.exists()
    
    # 验证配置内容
    new_config = json.loads(new_path.read_text())
    assert new_config["jobs"][0]["name"] == "morning_report"
```

#### TC-FEISHU-PERF-002: 并发处理性能

```python
import pytest
import concurrent.futures

def test_concurrent_message_handling():
    """测试并发消息处理"""
    from src.notify.feishu import FeishuBot
    
    bot = FeishuBot()
    
    messages = [f"并发测试消息 {i}" for i in range(20)]
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=5) as executor:
        results = list(executor.map(bot.send_message, messages))
    
    # 验证所有消息都成功发送
    success_count = sum(1 for r in results if r.get("success"))
    assert success_count >= len(messages) * 0.9  # 90%成功率
```

```python
"""飞书模块单元测试"""

import pytest
from unittest.mock import patch, MagicMock
from src.notify.feishu import FeishuBot


class TestFeishuBot:
    """飞书机器人测试类"""
    
    @pytest.fixture
    def bot(self):
        """创建飞书机器人实例"""
        with patch.dict('os.environ', {
            'FEISHU_APP_ID': 'test_app_id',
            'FEISHU_APP_SECRET': 'test_app_secret'
        }):
            return FeishuBot()
    
    def test_initialization(self, bot):
        """测试初始化"""
        assert bot.app_id == 'test_app_id'
        assert bot.app_secret == 'test_app_secret'
    
    def test_get_access_token(self, bot):
        """测试获取访问令牌"""
        with patch('requests.post') as mock_post:
            mock_post.return_value.status_code = 200
            mock_post.return_value.json.return_value = {
                'code': 0,
                'tenant_access_token': 'test_token',
                'expire': 7200
            }
            
            token = bot._get_access_token()
            
            assert token == 'test_token'
    
    def test_send_text_message(self, bot):
        """测试发送文本消息"""
        with patch.object(bot, '_get_access_token', return_value='test_token'):
            with patch('requests.post') as mock_post:
                mock_post.return_value.status_code = 200
                mock_post.return_value.json.return_value = {
                    'code': 0,
                    'data': {'message_id': 'test_msg_id'}
                }
                
                result = bot.send_text_message(
                    receive_id='test_user',
                    content='测试消息'
                )

                assert result['message_id'] == 'test_msg_id'
```
